CREATE FULLTEXT INDEX "TA"."MINING" ON "TA"."BEWERTUNGEN" ("STRING") FAST PREPROCESS OFF TEXT MINING ON LANGUAGE DETECTION ('DE');


SELECT SCORE, * FROM TM_GET_RELATED_DOCUMENTS(
	DOCUMENT (
		SELECT "STRING" FROM "TA"."BEWERTUNGEN"
		WHERE "ID" = 1
	)
	SEARCH DISTINCT "STRING" FROM "TA"."BEWERTUNGEN"
	RETURN PRINCIPAL COMPONENTS 2 CORRELATION TOP 20 "ID", "STRING"
) AS T;




SELECT T.TERM, T.NORMALIZED_TERM, T.TERM_TYPE, T.TERM_FREQUENCY, T.DOCUMENT_FREQUENCY, T.SCORE
FROM TM_GET_RELATED_TERMS(
	TERM 'burger'
	SEARCH "STRING" FROM "TA"."BEWERTUNGEN"
	RETURN TOP 16
) AS T;
